// polygon.js 파일
var globalMap;

function addPolygon(map, coordinates, overlayContent) {
    // 다각형 생성
    const polygon = new kakao.maps.Polygon({
        map: map, // 수정: map 변수를 사용
        path: coordinates,
        strokeWeight: 3,
        strokeColor: "red",
        strokeOpacity: 1,
        fillColor: 'none',
        fillOpacity: 0.5,
    });
}
